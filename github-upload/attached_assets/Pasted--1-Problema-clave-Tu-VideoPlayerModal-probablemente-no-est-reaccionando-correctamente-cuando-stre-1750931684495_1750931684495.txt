import { useEffect, useRef } from "react";

type VideoPlayerModalProps = {
  isOpen: boolean;
  onClose: () => void;
  streamUrl: string;
  title: string;
};

export function VideoPlayerModal({ isOpen, onClose, streamUrl, title }: VideoPlayerModalProps) {
  const videoRef = useRef<HTMLVideoElement | null>(null);

  // Recargar el video cuando cambia la URL
  useEffect(() => {
    if (isOpen && streamUrl && videoRef.current) {
      videoRef.current.load();
    }
  }, [streamUrl, isOpen]);

  if (!isOpen) return null;

  return (
    <div className="fixed inset-0 z-50 bg-black bg-opacity-80 flex items-center justify-center">
      <div className="bg-gray-900 p-4 rounded-xl w-full max-w-3xl">
        <h2 className="text-white text-xl mb-4">{title}</h2>
        <video
          ref={videoRef}
          controls
          autoPlay
          className="w-full rounded-lg"
        >
          <source src={streamUrl} type="video/mp4" />
